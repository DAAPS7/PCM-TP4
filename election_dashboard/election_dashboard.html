// Return the votes for all districts from the selected party
function partyVotesByDistrict(party) {
  const result = [];
  const partyData = tbl[party];

  for (let [Distrito, Votos] of Object.entries(partyData)) {
    // Make sure the "-" and "c.r" votes convert to 0
    Votos = parseInt(Votos) || 0;
    
    result.push({ Distrito, Votos });
  }

  return result.sort((a, b) => b.Votos - a.Votos);
}

// Return the the votes for all parties in the selected district
function districtVotes(districtName) {
  const result = [];

  for (const [Partido, Distritos] of Object.entries(tbl)) {
    if (Partido !== "Inscritos" && Partido !== "Votantes" && Partido !== "Abstenção" && Partido !== "Brancos" && Partido !== "Nulos" && Partido !== "Votos Val. Exp. (VVE)"
    ) {
      // Make sure the "-" and "c.r" votes convert to 0
      const Votos = parseInt(Distritos[districtName]) || 0;

      if (Votos !== undefined) {
        result.push({ Partido, Votos });
      }
    }
  }

  return result.sort((a, b) => b.Votos - a.Votos);
}

// Converts text to title case (First letter of each word capitalized)
function toTitleCase(str) {
  return str.toLowerCase().split(' ').map(word => {
    if(word !== "do")
      return word.charAt(0).toUpperCase() + word.slice(1);
    else
      return word;
  }).join(' ');
}

// Return the ranking of abstention
function getRanking(district) {
  const table = tbl["Abstenção"];

  // Remove the total and sorts the array from lowest to highest
  const sortedArray = Object.entries(table).filter(([district]) => district !== "Total").sort((a, b) => a[1] - b[1]);

  // Find the array index from the district name
  const ranking = sortedArray.findIndex(([name]) => name === district);

  // Sum 1 because of array indexing
  return ranking + 1;
}

function calculateIntensity(party, district) {
  // Party votes for the selected district
  const partyVotes = parseInt(tbl[party]?.[district]) || 0;
  // Total votes of all parties for the district
  const districtVotes = tbl["Votantes"]?.[district] || 1;

  return partyVotes / districtVotes;
}

function convertByParty() {
  const root = { name: "Portugal", children: [] };

  getParties().forEach(partido => {
    const partidoNode = { name: partido, children: [] };

    for (const [distrito, votos] of Object.entries(tbl[partido])) {
      if(distrito !== "Total")
        partidoNode.children.push({ name: distrito, value: votos });
    }

    root.children.push(partidoNode);
  });

  return root;
}

chart = {
  // Specify the chart’s dimensions.
  const width = 600;
  const height = width;
  const radius = width / 6;

  // Compute the layout.
  const hierarchy = d3.hierarchy(flareData)
      .sum(d => d.value)
      .sort((a, b) => b.value - a.value);

  const root = d3.partition()
      .size([2 * Math.PI, hierarchy.height + 1])
    (hierarchy);

  root.each(d => d.current = d);

  // Create the arc generator.
  const arc = d3.arc()
      .startAngle(d => d.x0)
      .endAngle(d => d.x1)
      .padAngle(d => Math.min((d.x1 - d.x0) / 2, 0.005))
      .padRadius(radius * 1.5)
      .innerRadius(d => d.y0 * radius)
      .outerRadius(d => Math.max(d.y0 * radius, d.y1 * radius - 1))

  // Create the SVG container.
  const svg = d3.create("svg")
      .attr("viewBox", [-width / 2, -height / 2, width, width])
      .style("font", "8px sans-serif");

  // Append the arcs.
  const path = svg.append("g")
    .selectAll("path")
    .data(root.descendants().slice(1))
    .join("path")
      .attr("fill", d => { while(d.depth > 1) d = d.parent; return getPartyColors()[d.data.name] || "#b3afafab"; })
      .attr("fill-opacity", d => arcVisible(d.current) ? (d.children ? 0.6 : 0.4) : 0)
      .attr("pointer-events", d => arcVisible(d.current) ? "auto" : "none")
      .attr("d", d => arc(d.current));

  // Make them clickable if they have children.
  path.filter(d => d.children)
      .style("cursor", "pointer")
      .on("click", clicked);

  const format = d3.format(",d");
  path.append("title")
      .text(d => `${d.ancestors().map(d => d.data.name).reverse().join("/")}\n${format(d.value)}`);

  const label = svg.append("g")
      .attr("pointer-events", "none")
      .attr("text-anchor", "middle")
      .style("user-select", "none")
    .selectAll("text")
    .data(root.descendants().slice(1))
    .join("text")
      .attr("dy", "0.35em")
      .attr("fill-opacity", d => +labelVisible(d.current))
      .attr("transform", d => labelTransform(d.current))
      .text(d => d.data.name);

  const parent = svg.append("circle")
      .datum(root)
      .attr("r", radius)
      .attr("fill", "none")
      .attr("pointer-events", "all")
      .on("click", clicked);

  // Handle zoom on click.
  function clicked(event, p) {
    parent.datum(p.parent || root);

    root.each(d => d.target = {
      x0: Math.max(0, Math.min(1, (d.x0 - p.x0) / (p.x1 - p.x0))) * 2 * Math.PI,
      x1: Math.max(0, Math.min(1, (d.x1 - p.x0) / (p.x1 - p.x0))) * 2 * Math.PI,
      y0: Math.max(0, d.y0 - p.depth),
      y1: Math.max(0, d.y1 - p.depth)
    });

    const t = svg.transition().duration(event.altKey ? 7500 : 750);

    // Transition the data on all arcs, even the ones that aren’t visible,
    // so that if this transition is interrupted, entering arcs will start
    // the next transition from the desired position.
    path.transition(t)
        .tween("data", d => {
          const i = d3.interpolate(d.current, d.target);
          return t => d.current = i(t);
        })
      .filter(function(d) {
        return +this.getAttribute("fill-opacity") || arcVisible(d.target);
      })
        .attr("fill-opacity", d => arcVisible(d.target) ? (d.children ? 0.6 : 0.4) : 0)
        .attr("pointer-events", d => arcVisible(d.target) ? "auto" : "none") 

        .attrTween("d", d => () => arc(d.current));

    label.filter(function(d) {
        return +this.getAttribute("fill-opacity") || labelVisible(d.target);
      }).transition(t)
        .attr("fill-opacity", d => +labelVisible(d.target))
        .attrTween("transform", d => () => labelTransform(d.current));
  }
  
  function arcVisible(d) {
    return d.y1 <= 3 && d.y0 >= 1 && d.x1 > d.x0;
  }

  function labelVisible(d) {
    return d.y1 <= 3 && d.y0 >= 1 && (d.y1 - d.y0) * (d.x1 - d.x0) > 0.03;
  }

  function labelTransform(d) {
    const x = (d.x0 + d.x1) / 2 * 180 / Math.PI;
    const y = (d.y0 + d.y1) / 2 * radius;
    return `rotate(${x - 90}) translate(${y},0) rotate(${x < 180 ? 0 : 180})`;
  }

  svg.attr("width", "100%").attr("height", "80vh");

  const sunburstDiv = html `<div style="height: 80vh; width: 100%;"> 
                ${svg.node()} </div>`;

  return sunburstDiv;
}

function getPartyVotes(municipality) {
  const result = [];

  for(const [Concelho, Partidos] of Object.entries(muniTbl)) {
    if(Concelho === municipality) {
      for(const[Partido, Votos] of Object.entries(Partidos)) {
        if(Partido !== "Brancos" && Partido !== "Nulos" && Partido !== "Votantes" && Partido !== "Inscritos") {
          result.push({ Partido, Votos });
        }
      }
    }
  }

  return result.sort((a, b) => b.Votos - a.Votos);
}

// Formats the municipalities strings to the correct format
function toTitleCaseMunicipalities(str) {
  if(str.includes("-")) {
    return str.toLowerCase().split('-').map(word => {
      if(word.length > 1)
        return word.charAt(0).toUpperCase() + word.slice(1);
      else
        return word;
  }).join('-');
  } else {
    return str.toLowerCase().split(' ').map(word => {
    if(word !== "do" && word !== "da" && word !== "de" && word !== "dos" && word !== "das" && word !== "à")
      return word.charAt(0).toUpperCase() + word.slice(1);
    else
      return word;
  }).join(' ');
  }
}

scatterPlot = {
  // Prepare data for scatter plot
  const scatterData = [];
  
  for (const [Concelho, data] of Object.entries(muniTbl)) {
    // Skip "Território Nacional"
    if (Concelho === "Território Nacional") continue;
    
    const inscritos = data["Inscritos"];
    const votantes = data["Votantes"];
    
    // Calculate abstention rate
    const abstencao = inscritos > 0 ? ((inscritos - votantes) / inscritos) * 100 : 0;
    
    // Get winning party
    const winner = getPartyVotes(Concelho)[0]?.Partido || "Unknown";
    
    scatterData.push({
      concelho: Concelho,
      inscritos: inscritos,
      abstencao: abstencao,
      partido: winner
    });
  }
  
  // Set up dimensions
  const width = 928;
  const height = 500;
  const marginTop = 20;
  const marginRight = 20;
  const marginBottom = 40;
  const marginLeft = 60;
  
  // Create scales
  const x = d3.scaleLinear()
    .domain([0, d3.max(scatterData, d => d.inscritos)])
    .nice()
    .range([marginLeft, width - marginRight]);
  
  const y = d3.scaleLinear()
    .domain([0, d3.max(scatterData, d => d.abstencao)])
    .nice()
    .range([height - marginBottom, marginTop]);

  const svg = d3.create("svg")
    .attr("width", width)
    .attr("height", height)
    .attr("viewBox", [0, 0, width, height])
